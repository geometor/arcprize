import numpy as np

def transform_grid(grid):
    """Transforms the input grid based on the observed rules."""
    new_grid = np.copy(grid)
    rows, cols = grid.shape

    #Find red and blue pixels
    red_coords = np.where(grid == 2)
    blue_coords = np.where(grid == 1)

    # Apply red pixel rule
    for r, c in zip(*red_coords):
        if r > 0 and c > 0:
            new_grid[r - 1, c - 1] = 4
        if r > 0 and c < cols - 1:
            new_grid[r - 1, c + 1] = 4
        if r < rows - 1 and c > 0:
            new_grid[r + 1, c - 1] = 4
        if r < rows - 1 and c < cols -1:
            new_grid[r + 1, c + 1] = 4

    # Apply blue pixel rule
    for r, c in zip(*blue_coords):
        if r > 0:
            new_grid[r - 1, c] = 7
        if r < rows - 1:
            new_grid[r + 1, c] = 7
        if c > 0:
            new_grid[r, c - 1] = 7
        if c < cols - 1:
            new_grid[r, c + 1] = 7

    return new_grid
