def count_exterior_edges(grid):
    rows, cols = grid.shape
    count = 0
    for i in range(rows):
        for j in range(cols):
            if grid[i, j] == 8:
                neighbors = []
                for x, y in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
