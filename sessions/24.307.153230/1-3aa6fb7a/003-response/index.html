import numpy as np

def is_isolated(grid, row, col):
    rows, cols = grid.shape
    if grid[row, col] != 8:
        return False
    for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
        nr, nc = row + dr, col + dc
        if 0 <= nr < rows and 0 <= nc < cols:  #Check boundaries explicitly
            if grid[nr, nc] == 8:
                return False
    return True

def transform_grid(grid):
    rows, cols = grid.shape
    new_grid = np.copy(grid)
    for r in range(rows):
        for c in range(cols):
            if is_isolated(grid, r, c):
                new_grid[r, c] = 1
    return new_grid
