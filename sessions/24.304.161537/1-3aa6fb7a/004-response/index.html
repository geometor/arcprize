def transform_grid(grid):
    rows = len(grid)
    cols = len(grid[0]) if rows > 0 else 0
    new_grid = [row[:] for row in grid]  # Create a copy to avoid modifying the original

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 8:
                if r > 0 and c < cols - 1 and grid[r - 1][c + 1] == 0:
                    new_grid[r - 1][c + 1] = 1
    return new_grid
