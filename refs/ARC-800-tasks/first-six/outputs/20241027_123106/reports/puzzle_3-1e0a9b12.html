import numpy as np

def summarize_grid(grid):
    """Summarizes a grid by counting colors and finding objects."""
    height, width = grid.shape
    color_counts = {}
    for color in range(10):
        color_counts[color] = np.sum(grid == color)

    objects = []
    for row in range(height):
        for col in range(width):
            if grid[row, col] != 0:
                # Find the extent of the object
                object_size = 1
                for i in range(col + 1, width):
                    if grid[row, i] == grid[row, col]:
                        object_size += 1
                    else:
                        break
                objects.append({
                    'color': grid[row, col],
                    'size': object_size,
                    'position': (row, col)
                })

    return {
        'width': width,
        'height': height,
        'colors': color_counts,
        'objects': objects
    }
